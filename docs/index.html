# filter out Delaware county 
df_delaware <- df_clean |>
  filter(geo_name == "Delaware") |>
  left_join(
    df_clean |>
      filter(geo_name == "Delaware", line_code == "1") |>
      select(time_period, total_gdp = data_value),
    by = "time_period"
  ) |>
  mutate(share = data_value / total_gdp) |>
  filter(line_code %in% c(3,6,10,11,12,34,35,36,45,51,56,60,64,65,69,70,76,79,82,83))|>
  select(geo_name,line_code,time_period,description, share)

# AI-assisted
# calculate average share during 2001-2024, tag "other" label
industry_avg <- df_delaware |>
  group_by(line_code, description) |>
  summarise(avg_share = mean(share), .groups = "drop")

# share >= 0.05
major_industries <- industry_avg |>
  filter(avg_share >= 0.05) |>
  pull(description)

df_delaware <- df_delaware |>
  mutate(description_grouped = if_else(description %in% major_industries, description, "Other"))

df_delaware_wide <- df_delaware |>
  group_by(time_period, description_grouped) |>
  summarise(share = sum(share), .groups = "drop") |>
  pivot_wider(names_from = description_grouped, values_from = share)

write_csv(df_delaware_wide, "../data/processed/delaware_industry_2001-2024.csv")

# Mutate change and percantage change number
del_all_flow_wide <- del_all_flow |>
  filter(year %in% c(2013, 2023)) |>
  pivot_wider(names_from = year, values_from = workers, names_prefix = "workers_") |>
  mutate(
    change = workers_2023 - workers_2013,
    pct_change = (workers_2023 - workers_2013) / workers_2013 * 100
  ) |>
  arrange(desc(change))
